f_a_cartesian_product(a_a, a_b):
    # a_a = [[2,3], [3,2], [22,3], ...]
    # a_b = [[33,12], [54,23], [2,1], ...]
    # a_product = [[2,3,33,12],[2,3,54,23],[2,3,2,1],[3,2,33,12]...]
    a_product = []
    for a_a_a in a_a:
        for a_a_b in a_b: 
            a_product.append(a_a_a + a_a_b)

    return a_product
